print(3 in s1)

print(10 not in s1)

s1={3,4,5}
s2={4,5,6,7}
s3=s1&s2 # 交集︰取兩個集合中，相同的資料
print(s3)
s3=s1|s2 # 聯集︰取兩個合中的所有資料，但不重複取
print(s3)
s3=s1-s2 # 差集︰從 s1 中，減去和 s2 重疊的部分
print(s3)
s3=s1^s2 # 反交集︰取兩個集合中，重複的部分
print(s3)

{3}

#字典的運用︰key-value 配對

print(dic)
print(dic["apple"])

print(dic["apple"])
